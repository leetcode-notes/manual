def euler(q,p,eps,lam,k,f):
    q_new = q + eps * p
    p_new = p + eps * f(q, lam, k)
    q, p = q_new, p_new
    return q, p

def leap_flog(q,p,eps,lam,k,f):
    p_new = p + 0.5 * eps * f(q, lam, k)
    q_new = q + eps * p_new
    p_new = p_new + 0.5 * eps * f(q_new,lam, k)
    q, p = q_new, p_new
    return q, p
