# numpy 条件に合った値をとってくる

# 早くないやり方
A = np.array([1,2,3,4])
B= np.array([100,200,300,400])

# 真偽値のアレイ
condition = np.array([True,True,False,False])

# リスト内包表記を使った例
answer = [(a if cond else b) for a,b,cond in zip(A,B,condition)]
