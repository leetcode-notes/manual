# 正規分布に従う乱数です。
data1 = stats.norm(0,5).rvs(100)

# γ分布に従う乱数を生成します。
data2 = np.concatenate([stats.gamma(5).rvs(50)-1,
                        -1*stats.gamma(5).rvs(50)])

sns.boxplot(data=[data1,data2],whis=np.inf)
