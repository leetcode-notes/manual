// 西暦年から平成年を求める
using System;
public class Program{
    public static void Main(){
        var today = DateTime.Today;
        var seireki = today.Year;
        Console.Write("西暦" + seireki + "年は、");
        
        var heisei = seireki - 1988;
        Console.WriteLine("平成" + heisei + "年です。");
    }
}
