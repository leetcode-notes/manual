using System;

public class Lesson05
{
    public static void Main()
    {
        string[][] teams = {
            new string[] {"勇者", "戦士", "魔法使い"},
            new string[] {"盗賊", "忍者", "商人"},
            new string[] {"スライム", "ドラゴン", "魔王"},
            new string[] {"踊り子", "僧侶", "アーチャー"}
        };

        for (int i = 0; i < teams.Length; i++)
        {
            for (int j = 0; j < teams[i].Length; j++)
            {
                Console.Write(teams[i][j] + " ");
            }
            Console.WriteLine();
        }
    }
}
