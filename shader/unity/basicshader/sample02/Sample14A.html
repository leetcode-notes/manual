        _Roughness     ("Roughness", Range(0, 1.0)) = 0
        _Fresnel       ("Fresnel",   Range(0, 1.0)) = 0
    }

    SubShader
    {
        Pass
        {
            Tags
            {
                "LightMode" = "ForwardBase"
            }

            CGPROGRAM

            #pragma vertex   vert
            #pragma fragment frag

            #include "Sample14ALib.cginc"

            ENDCG
        }

        {
            Tags
            {
                "LightMode" = "ForwardAdd"
            }

            CGPROGRAM

            #pragma fragment frag

            #include "Sample14ALib.cginc"

            ENDCG
        }
    }
}
