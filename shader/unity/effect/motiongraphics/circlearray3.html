motiongraphics circlearray3¶

Shader "Custom/Circle" {
	Properties {
		_Color ("Color", Color) = (1,1,1,1)
		_MainTex ("Albedo (RGB)", 2D) = "white" {}
		_Glossiness ("Smoothness", Range(0,1)) = 0.5
		_Metallic ("Metallic", Range(0,1)) = 0.0
		_Power("Power", Range(0, 1)) = 0.0
	}
	SubShader {
		Tags { "RenderType"="Opaque" }
		LOD 200

		CGPROGRAM
		#pragma surface surf Standard vertex:vert nolightmap addshadow

		#pragma target 3.0

		#define PI 3.1415926535

		struct Input {
			float4 color: COLOR;
			float2 uv_MainTex: TEXCOORD0;
		};

		half _Glossiness;
		half _Metallic;
		fixed4 _Color;
		float _Power;

		float rand(float x, float y) {
			return frac(sin(dot(float2(x, y), float2(12.9898, 78.233))) * 43758.5453);
		}

		void vert(inout appdata_full v) {
			float3 vertex = v.vertex.xyz;
			float2 texcoord = v.texcoord;

			vertex.xyz /= 2.2;
			vertex.z *= PI * 2 / 50;

			float yPos = 0;
			yPos += clamp(sin(texcoord.x * PI * 4 + _Time.y * sin(rand(texcoord.y, 0)) * 4 + rand(texcoord.y, 1) * 5) - 0.8, 0, 1) * 10;
			yPos += rand(texcoord.y, texcoord.x) * _Power * 5;


			vertex.y += yPos;
			vertex.x += 1 + texcoord.y * 50;
			vertex.z *= vertex.x;

			float angle = texcoord.x * PI * 2;
			vertex.xz = float2(
				cos(angle) * vertex.x + sin(angle) * vertex.z,
				-sin(angle) * vertex.x + cos(angle)  * vertex.z);

			v.vertex.xyz = vertex;

			float colorSeed = rand(texcoord.x, texcoord.y);
			if (colorSeed < 0.6) v.color.rgb = float3(1, 1, 1);
			else if (colorSeed < 0.8) v.color.rgb = float3(1, 0, 0);
			else v.color.rgb = float3(0, 0, 0);

			v.color.a = 1;
		}

		void surf (Input IN, inout SurfaceOutputStandard o) {
			fixed4 c = IN.color;
			o.Albedo = c.rgb;
			o.Metallic = _Metallic;
			o.Smoothness = _Glossiness;
			o.Alpha = c.a;
		}
		ENDCG
	}
	FallBack "Diffuse"
}
