StageColor StageColorTrack¶

using UnityEngine;
using UnityEngine.Playables;
using UnityEngine.Timeline;

namespace Inspix.Timeline
{
    /// <summary>
    /// ステージの色変えをするタイムライン
    /// </summary>
    [TrackColor(0.4100659f, 0.8962264f, 0.8861892f)]
    [TrackClipType(typeof(StageColorClip))]
    public class StageColorTrack : TrackAsset
    {
        public override Playable CreateTrackMixer(PlayableGraph graph, GameObject go, int inputCount)
        {
            return ScriptPlayable<StageColorMixerBehaviour>.Create(graph, inputCount);
        }
    }
}
