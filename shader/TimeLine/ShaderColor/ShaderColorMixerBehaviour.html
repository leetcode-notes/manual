ShaderColor ShaderColorMixerBehaviour¶

using System;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.Playables;

namespace Inspix.Timeline
{
    /// <summary>
    /// ShaderのColorを変更するタイムライン（MixerBehaviour）
    /// </summary>
    public class ShaderColorMixerBehaviour : PlayableBehaviour
    {
        List<MeshRenderer> targetRenderers = new List<MeshRenderer>();
        List<Color> defaultColors = new List<Color>();
        MaterialPropertyWrapper block;

        static readonly int BASE_COLOR_ID = Shader.PropertyToID("_Color");


        public void Initialize(String objectName, bool isApplyAllChildren)
        {
            if (this.block == null)
            {
                this.block = new MaterialPropertyWrapper();
            }
            this.targetRenderers.Clear();
            this.defaultColors.Clear();

            var sceneGameObjects = GameObjectUtil.GetHierarcheyObjects();
            var obj = sceneGameObjects.Find(o => o.name == objectName);
            if (!obj)
            {
                return;
            }

            if (isApplyAllChildren)
            {
                initializeAllChildren(obj);
            }
            else
            {
                initializeSingleObject(obj);
            }
        }

        void initializeSingleObject(GameObject obj)
        {
            var mesh = obj.GetComponent<MeshRenderer>();
            if (!mesh)
            {
                return;
            }

            if (!mesh.sharedMaterial.HasProperty(BASE_COLOR_ID))
            {
                return;
            }

            this.targetRenderers.Add(mesh);
            this.defaultColors.Add(mesh.sharedMaterial.GetColor(BASE_COLOR_ID));
        }

        void initializeAllChildren(GameObject obj)
        {
            foreach (Transform child in obj.transform.GetAllChildren())
            {
                var mesh = child.GetComponent<MeshRenderer>();
                if (!mesh)
                {
                    continue;
                }

                if (!mesh.sharedMaterial.HasProperty(BASE_COLOR_ID))
                {
                    continue;
                }

                this.targetRenderers.Add(mesh);
                var color = mesh.sharedMaterial.GetColor(BASE_COLOR_ID);
                this.defaultColors.Add(color);
            }
        }

        public override void ProcessFrame(Playable playable, FrameData info, object playerData)
        {
            if (this.targetRenderers.Count <= 0)
            {
                return;
            }

            int inputCount = playable.GetInputCount();
            var totalColor = Color.black;
            for (int i = 0; i < inputCount; ++i)
            {
                float inputWeight = playable.GetInputWeight(i);
                if (inputWeight > 0.001f)
                {
                    var inputPlayable = (ScriptPlayable<ShaderColorBehaviour>)playable.GetInput(i);
                    var input = inputPlayable.GetBehaviour();
                    totalColor += (input.Color * inputWeight);
                }
            }
            this.block.SetColor(BASE_COLOR_ID, totalColor);
            foreach (var renderer in this.targetRenderers)
            {
                renderer.SetPropertyBlock(this.block);
            }

        }

        public override void OnGraphStop(Playable playable)
        {
            if (this.targetRenderers.Count <= 0)
            {
                return;
            }

            for (int i = 0; i < this.targetRenderers.Count; i++)
            {
                this.block.SetColor(BASE_COLOR_ID, this.defaultColors[i]);
                this.targetRenderers[i].SetPropertyBlock(this.block);
            }
        }
    }
}
