    struct appdata
    {
        float4 vertex : POSITION;
        half3 normal : NORMAL;
        float2 uv : TEXCOORD0;
        float4 tangent : TANGENT;
        UNITY_VERTEX_INPUT_INSTANCE_ID
    };

    struct v2f
    {
        float4 pos : SV_POSITION;
        float2 uv : TEXCOORD0;
        half3 normal : TEXCOORD1;
        half3 eyeDir : TEXCOORD2;
        half3 lightDir : TEXCOORD3;
    #ifdef _IS_SPECULARMASK
       half3 halfDir : TEXCOORD4;
    #endif
    #ifdef _IS_ANGELRING
        half3 viewforward : TEXCOORD6;
    #endif
        half3 tspace0 : TEXCOORD7;
        half3 tspace1 : TEXCOORD8;
        half3 tspace2 : TEXCOORD9;
        float3 posWorld : TEXCOORD10;
        UNITY_VERTEX_INPUT_INSTANCE_ID
        UNITY_VERTEX_OUTPUT_STEREO
    };

    sampler2D _MainTex;
    sampler2D _SColor;  

#ifdef _IS_RIMLIGHT
    //リムライト
    sampler2D _RimMaskTex;
#endif


    CBUFFER_START(UnityPerMaterial)
    float4 _MainTex_ST;

    //リムライト
    float4 _RimMaskTex_ST;
    uniform half4 _RimColor;
    uniform float _RimWidth;
    uniform float _RimPower;
    uniform float _RimRearAdjustment;
    uniform float _RimBias;

    float _Cutoff;
    
    //2color
    float _RampThreshold;
    float _RampSmoothing;
    half4 _SColor2;
    
    CBUFFER_END

    v2f vert (appdata v)
    {
        v2f o = (v2f)0;

        UNITY_SETUP_INSTANCE_ID(v);
        UNITY_TRANSFER_INSTANCE_ID(v, o);
        UNITY_INITIALIZE_VERTEX_OUTPUT_STEREO(o);

        VertexPositionInputs vertexInput = GetVertexPositionInputs(v.vertex.xyz);

        VertexNormalInputs normalInput = GetVertexNormalInputs(v.normal);
        o.pos = vertexInput.positionCS;
        o.posWorld = vertexInput.positionWS;
        o.normal = NormalizeNormalPerVertex(normalInput.normalWS);
        o.eyeDir = normalize(GetCameraPositionWS() - vertexInput.positionWS);

        half3 worldNormal = TransformObjectToWorldNormal(v.normal);
        half3 worldTangent = TransformObjectToWorldDir(v.tangent);
        half tangentSign = v.tangent.w * unity_WorldTransformParams.w;
        half3 worldBitangent = cross(worldNormal, worldTangent) * tangentSign;
        o.tspace0 = half3(worldTangent.x, worldBitangent.x, worldNormal.x);
        o.tspace1 = half3(worldTangent.y, worldBitangent.y, worldNormal.y);
        o.tspace2 = half3(worldTangent.z, worldBitangent.z, worldNormal.z);
        
        o.uv = v.uv; // そのままのTexcoord

        return o;
    }

    float4 frag (v2f i) : SV_Target
    {
        UNITY_SETUP_INSTANCE_ID(i);
        UNITY_SETUP_STEREO_EYE_INDEX_POST_VERTEX(i);

        Light mainLight = GetMainLight();
        half3 lightDir = mainLight.direction;

        half3 lightColor = mainLight.color.rgb;

        half4 col = tex2D( _MainTex, TRANSFORM_TEX( i.uv, _MainTex ) );
        
        //2color
        float _ShadeShift = ( _RampThreshold );
        float _ShadeToony = ( _RampSmoothing );
        half3 worldNormal = half3(i.tspace0.z, i.tspace1.z, i.tspace2.z);
        worldNormal *= step(0, dot(_WorldSpaceCameraPos.xyz - i.posWorld.xyz, worldNormal)) * 2 - 1;
        //worldNormal *= lerp(+1.0, -1.0, i.isOutline);
        worldNormal = normalize(worldNormal);
    
        // lighting intensity
        half receiveShadow = tex2D( _SColor, TRANSFORM_TEX( i.uv, _MainTex ) ).a;
        //half receiveShadow = tex2D(_SColor, mainUv).a;
        half atten = mainLight.distanceAttenuation * mainLight.shadowAttenuation;
        half lightIntensity = dot(mainLight.direction, worldNormal);
        lightIntensity = lightIntensity * 0.5 + 0.5; // from [-1, +1] to [0, 1]
        lightIntensity = lightIntensity * (1.0 - receiveShadow * (1.0 - (atten * 0.5 + 0.5))); // receive shadow
        lightIntensity = lightIntensity * 2.0 - 1.0; // from [0, 1] to [-1, +1]
        lightIntensity = smoothstep(_ShadeShift, _ShadeShift + (1.0 - _ShadeToony), lightIntensity); // shade & tooned
    
        // lighting with color
        half3 directLighting = mainLight.color; // direct

        // color lerp
        half4 shade = tex2D( _SColor, TRANSFORM_TEX( i.uv, _MainTex ) );
        half4 lit = tex2D( _MainTex, TRANSFORM_TEX( i.uv, _MainTex ) );
        half3 col2 = lerp(shade.rgb, lit.rgb, lightIntensity);
        col.rgb = col2;

        //ライト色を反映
        col.rgb *= lightColor;

        //リムライト
#ifdef _IS_RIMLIGHT
        half3 ViewNormalDir = normalize(mul((float3x3)UNITY_MATRIX_V, i.normal));
        half4 rimcol = tex2D( _RimMaskTex, i.uv);
        float rim = 0.0;
        float bias = 0.0;

    #ifdef _IS_RIMBIAS
        bias = _RimBias;
    #endif

        //リムライトを裏側に描画しない
        if(dot(i.eyeDir, i.normal) >= -_RimRearAdjustment)
        {
            rim = 1.0 - saturate(dot (float3(i.eyeDir.x + bias, i.eyeDir.y, i.eyeDir.z), i.normal));
        }

        half4 color = 0;
    #ifdef _IS_LIGHTCOLOR
        //リムライトの色をディクショナルライトの色にする。
        color.rgb = smoothstep(_RimWidth, _RimPower, rim) * lightColor * rimcol.r;
    #else
        //リムライトの色を指定色にする。
        color.rgb = smoothstep(_RimWidth, _RimPower, rim) * lightColor * _RimColor * rimcol.r;
    #endif

        col.rgb += color.rgb;
#endif

        clip( col.a - _Cutoff);

        /// END ///
        return col;
   }
