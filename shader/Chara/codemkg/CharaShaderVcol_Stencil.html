codemkg CharaShaderVcol_Stencil¶

Shader "Chara/CharaShaderVcol_Stencil"
{
    Properties
    {
        [HideInInspector] _Mode ("__mode", Float) = 1.0

        [Enum(UnityEngine.Rendering.CullMode)]_CullMode("Cull Mode", Float) = 2
        [Enum(UnityEngine.Rendering.CompareFunction)]_ZTestMode("ZTest Mode", Float) = 4
        [Toggle]_ZWriteParam("ZWrite", Float) = 1
        [HideInInspector] _BlendMode ("__blendmode", Float) = 0.0
        [Enum(UnityEngine.Rendering.BlendMode)]_SrcBlend("Blend Src", Float) = 5
        [Enum(UnityEngine.Rendering.BlendMode)]_DstBlend("Blend Dst", Float) = 10

		_StencilComp("Stencil Comparison", Float) = 8
		_StencilNo("Stencil No", Float) = 1
		_StencilOpPass("Stencil Operation", Float) = 0
		_StencilOpFail("Stencil Operation", Float) = 0

        [Toggle] _Is_Shadow("Is_Shadow", Float) = 0

        _Cutoff("Alpha Cutoff", Range(0.0, 1.0)) = 0.5
        [HideInInspector][Toggle] _UseCutout("__useCutout", Float) = 0

        _MainTex ("Main", 2D) = "black" {}

        //リムライト
        [Toggle(_IS_RIMLIGHT)]
        _Is_RimLight ("Is_RimLight", Float ) = 0
        _RimMaskTex ("RimMaskTex", 2D) = "white" {}
        [Toggle(_IS_LIGHTCOLOR)]
        _Is_LightColor ("Is_LightColor", Float ) = 1
        _RimColor ("RimColor", Color) = (1, 1, 1, 1)
        _RimWidth("RimWidth", Range(0, 1)) = 0
        _RimPower("RimPower", Range(0, 1)) = 1
        _RimRearAdjustment("RimRearAdjustment", Range(0, 1)) = 0.1
        [Toggle(_IS_RIMBIAS)]
        _Is_RimBias ("Is_RimBias", Float ) = 0
        _RimBias("RimBias", Range(-1, 1)) = 0

        //頂点カラー頬染め
        //一時オンオフ切り替え無し、オフにすると描画優先がかわってしまう？
        [Toggle(_IS_VERTEXCOLORMULTI)]
        _Is_VertexColorMulti ("_Is_VertexColorMulti", Float ) = 0
        _VertexColor ("VcolColor", Color ) =  (1, 1, 1, 1)
        _VcolPower("VcolPower", Range(0, 1)) = 1

        //瞳ウルウル
        //_MainTex ("Texture", 2D) = "white" {}
        _Flat ("_Flat", Range(0, 1)) = 1
        _Scale ("_Scale", Range(0, 0001)) = 0

        //アウトライン
        [HideInInspector]_OutlineMaskTex ("OutlineMaskTex", 2D) = "white" {}
        [HideInInspector]_OutlineMainColor ("OutlineMainColor", Color) = (.5,.5,.5,1)
        [HideInInspector]_OutlineColor ("OutlineColor", Color) = (0, 0, 0, 1)
        [HideInInspector]_Outline  ("Outline", float) = .005
        [HideInInspector]_OutlineMax("OutlineMax", Range(0.0, 10.0)) = .1
        [HideInInspector][Toggle(OUTLINEZWRITE)]_OutlineZWrite("OutlineZWrite", Float) = 1
        // add mm Zwrite
        [HideInInspector] _ZWrite ("_ZWrite", Float) = 1.0
    }
    SubShader
    {
        // Forward Base
        Pass
        {
            Name "FORWARD_BASE"
            Tags { "LightMode" = "UniversalForward" }


            Cull [_CullMode]
            ZTest [_ZTestMode]
            ZWrite [_ZWriteParam]
            Blend [_SrcBlend] [_DstBlend]

            //ステンシルバッファに指定値を書き込む
            Stencil
            {
                Ref[_StencilNo]
                Comp[_StencilComp]
                Pass[_StencilOpPass]
                Fail[_StencilOpFail]

            }

            HLSLPROGRAM

            #include "Packages/com.unity.render-pipelines.universal/ShaderLibrary/Lighting.hlsl"

            #pragma vertex vert
            #pragma fragment frag
            // GPU Instancing
            #pragma multi_compile_instancing

            #pragma shader_feature _ _USECUTOUT_ON
            #pragma shader_feature _ _IS_SPECULAR
            #pragma shader_feature _ _IS_RIMLIGHT
            #pragma shader_feature _ _IS_ANGELRING
            #pragma shader_feature _ _IS_LIGHTCOLOR
            #pragma shader_feature _ _IS_GRADATION
            #pragma shader_feature _ _IS_RIMBIAS
            #pragma shader_feature _ _IS_ANIMELINE

            #pragma shader_feature _ _IS_SPECULARMASK
            #pragma shader_feature _ _IS_VERTEXCOLORMULTI

            struct appdata
            {
                float4 vertex : POSITION;
                half3 normal : NORMAL;
                float2 uv : TEXCOORD0;
                float3 color: COLOR;
                UNITY_VERTEX_INPUT_INSTANCE_ID
            };

            struct v2f
            {
                float4 pos : SV_POSITION;
                float2 uv : TEXCOORD0;
                float3 color: COLOR;
                half3 normal : TEXCOORD1;
                half3 eyeDir : TEXCOORD2;
                half3 lightDir : TEXCOORD3;
            #ifdef _IS_SPECULARMASK
                half3 halfDir : TEXCOORD4;
            #endif
            #ifdef _IS_ANGELRING
                half3 viewforward : TEXCOORD6;
            #endif
                UNITY_VERTEX_INPUT_INSTANCE_ID
                UNITY_VERTEX_OUTPUT_STEREO
            };

            sampler2D _MainTex;



        #ifdef _IS_RIMLIGHT
            //リムライト
            sampler2D _RimMaskTex;
        #endif

            CBUFFER_START(UnityPerMaterial)
            float4 _MainTex_ST;

            //リムライト
            float4 _RimMaskTex_ST;
            uniform half4 _RimColor;
            uniform float _RimWidth;
            uniform float _RimPower;
            uniform float _RimRearAdjustment;
            uniform float _RimBias;

            float _Cutoff;

            //頂点カラー
            uniform half4 _Color;
            uniform float _VcolPower;
            CBUFFER_END

            v2f vert (appdata v)
            {
                v2f o = (v2f)0;

                UNITY_SETUP_INSTANCE_ID(v);
                UNITY_TRANSFER_INSTANCE_ID(v, o);
                UNITY_INITIALIZE_VERTEX_OUTPUT_STEREO(o);

                VertexPositionInputs vertexInput = GetVertexPositionInputs(v.vertex.xyz);
                VertexNormalInputs normalInput = GetVertexNormalInputs(v.normal);
                o.pos = vertexInput.positionCS;
                o.normal = NormalizeNormalPerVertex(normalInput.normalWS);
                o.eyeDir = normalize(GetCameraPositionWS() - vertexInput.positionWS);
                o.uv = v.uv;

                // 頂点カラー渡す
            #ifdef _IS_VERTEXCOLORMULTI
                o.color = v.color;
            #endif

                o.uv = v.uv; // そのままのTexcoord

                return o;
            }

            float4 frag (v2f i) : SV_Target
            {
                UNITY_SETUP_INSTANCE_ID(i);
                UNITY_SETUP_STEREO_EYE_INDEX_POST_VERTEX(i);

                Light mainLight = GetMainLight();
                half4 LightColor0 = half4(mainLight.color, 1);
                half4 col = tex2D( _MainTex, TRANSFORM_TEX( i.uv, _MainTex ) );

                //ライト色を反映
                col.rgb *= LightColor0.rgb;

                //リムライト
        #ifdef _IS_RIMLIGHT

                half3 ViewNormalDir = normalize(mul((float3x3)UNITY_MATRIX_V, i.normal));
                half4 rimcol = tex2D( _RimMaskTex, i.uv);
                float rim = 0.0;
                float bias = 0.0;

            #ifdef _IS_RIMBIAS
                bias = _RimBias;
            #endif

                //リムライトを裏側に描画しない
                if(dot(i.eyeDir, i.normal) >= -_RimRearAdjustment)
                {
                    rim = 1.0 - saturate(dot (float3(i.eyeDir.x + bias, i.eyeDir.y, i.eyeDir.z), i.normal));
                }

                half4 color = 0;
            #ifdef _IS_LIGHTCOLOR
                //リムライトの色をディクショナルライトの色にする。
                color.rgb = smoothstep(_RimWidth, _RimPower, rim) * LightColor0.rgb * rimcol.r;
            #else
                //リムライトの色を指定色にする。
                color.rgb = smoothstep(_RimWidth, _RimPower, rim) * LightColor0.rgb * _RimColor * rimcol.r;
            #endif

                col.rgb += color.rgb;
        #endif


            #ifdef _USECUTOUT_ON
                clip( col.a - _Cutoff);
            #endif

                //頂点カラー乗算
            #ifdef _IS_VERTEXCOLORMULTI
                half3 AddVcol =  saturate(i.color+_VcolPower);
                col.rgb *= AddVcol;
                //col.a = 1;
            #endif

                /// END ///
                return col;
            }
            ENDHLSL
        }
    }
