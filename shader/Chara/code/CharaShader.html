Properties {
		[HideInInspector] _Mode ("__mode", Float) = 1.0
		
		[Enum(UnityEngine.Rendering.CullMode)]_CullMode("Cull Mode", Float) = 2
		[Enum(UnityEngine.Rendering.CompareFunction)]_ZTestMode("ZTest Mode", Float) = 4
		[Toggle]_ZWriteParam("ZWrite", Float) = 1
		[HideInInspector] _BlendMode ("__blendmode", Float) = 0.0
		[Enum(UnityEngine.Rendering.BlendMode)]_SrcBlend("Blend Src", Float) = 5
		[Enum(UnityEngine.Rendering.BlendMode)]_DstBlend("Blend Dst", Float) = 10

		_StencilRef("StencilRef", Int) = 0

		 [Toggle] _Is_Shadow("Is_Shadow", Float) = 0

		_Cutoff("Alpha Cutoff", Range(0.0, 1.0)) = 0.5
		[HideInInspector][Toggle] _UseCutout("__useCutout", Float) = 0

		_MainTex ("Main", 2D) = "black" {}

		//パララックスマップ
		[Toggle(_IS_PARALLAXMAP)]
		_Is_ParallaxMap ("_Is_ParallaxMap", Float ) = 0
		[Toggle(_IS_PARALLAXMAP_BACK)]
		_Is_ParallaxMap_Back ("_Is_ParallaxMap_Back", Float ) = 0
		_ParallaxTex_Back ("ParallaxTex_Back", 2D) = "white" {}
		_ParallaxMaskTex_Back ("ParallaxMaskTex_Back", 2D) = "white" {}
		_ParallaxNormalMap_Back ("ParallaxNormalMap_Back", 2D) = "white" {}
        _ParallaxHeightMap_Back ("ParallaxHeightMap_Back", 2D) = "white" {}
		_HeightAdjustment_Back ("HeightAdjustment_Back", Range(0.0, 0.5)) = 0.02
		_PseudoLight_Back ("PseudoLight_Back", Vector) = (0, 0, 0)
		[Toggle(_IS_PARALLAXMAP_FRONT)]
		_Is_ParallaxMap_Front ("_Is_ParallaxMap_Front", Float ) = 0
		_ParallaxTex_Front ("ParallaxTex_Front", 2D) = "white" {}
		_ParallaxMaskTex_Front ("ParallaxMaskTex_Front", 2D) = "white" {}
		_ParallaxNormalMap_Front ("ParallaxNormalMap_Front", 2D) = "white" {}
        _ParallaxHeightMap_Front ("ParallaxHeightMap_Front", 2D) = "white" {}
		_HeightAdjustment_Front ("HeightAdjustment_Front", Range(0.0, 0.5)) = 0.02
		_PseudoLight_Front ("PseudoLight_Front", Vector) = (0, 0, 0)
		
		//グラデーション
		[Toggle(_IS_GRADATION)]
		_Is_Gradation ("Is_Gradation", Float ) = 0
		_GradationTex("Gradation", 2D) = "black" {}
		_GradationPower ("GradationPower", Range(0, 10)) = 1
		_SubtractionColor("SubtractionColor", Color) = (0, 0, 0, 0)

		//リムライト
		[Toggle(_IS_RIMLIGHT)]
		_Is_RimLight ("Is_RimLight", Float ) = 0
		_RimMaskTex ("RimMaskTex", 2D) = "white" {}
		[Toggle(_IS_LIGHTCOLOR)]
		_Is_LightColor ("Is_LightColor", Float ) = 1
        _RimColor ("RimColor", Color) = (1, 1, 1, 1)
		_RimWidth("RimWidth", Range(0, 1)) = 0

    SubShader
    {
		Tags {"IgnoreProjector"="True" "LightMode"="ForwardBase" }
		Cull [_CullMode]
		ZTest [_ZTestMode]
		ZWrite [_ZWriteParam]
		Blend [_SrcBlend] [_DstBlend]

        Pass
        {
			Name "BASE"

			//ステンシルバッファに指定値を書き込む
			Stencil
            {
				Ref [_StencilRef]
                Comp Always
                Pass Replace
            }
			
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
			#pragma shader_feature _ _USECUTOUT_ON
			#pragma shader_feature _ _IS_SPECULAR
			#pragma shader_feature _ _IS_RIMLIGHT
			#pragma shader_feature _ _IS_ANGELRING
			#pragma shader_feature _ _IS_LIGHTCOLOR
			#pragma shader_feature _ _IS_GRADATION
			#pragma shader_feature _ _IS_RIMBIAS
			#pragma shader_feature _ _IS_ANIMELINE
			#pragma shader_feature _ _IS_PARALLAXMAP
			#pragma shader_feature _ _IS_PARALLAXMAP_BACK
			#pragma shader_feature _ _IS_PARALLAXMAP_FRONT
			#pragma shader_feature _ _IS_SPECULARMASK

            #include "UnityCG.cginc"
			#include "AutoLight.cginc"
            struct appdata
            {
                float4 vertex : POSITION;
                half3 normal : NORMAL;
				float2 uv : TEXCOORD0;
				#ifdef _IS_PARALLAXMAP
				float4 tangent : TANGENT;
				#endif
            };

            struct v2f
            {
                float4 pos : SV_POSITION;
				float2 uv : TEXCOORD0;
                half3 normal : TEXCOORD1;
				#ifdef _IS_SPECULARMASK
                half3 halfDir : TEXCOORD3;
				#endif
				half3 eyeDir : TEXCOORD2;
				half3 lightDir : TEXCOORD4;
				#ifdef _IS_PARALLAXMAP
				half3 objeyeDir : TEXCOORD5;
				#endif
				#ifdef _IS_ANGELRING
				half3 viewforward : TEXCOORD6;
				#endif
            };
            
			sampler2D _MainTex;
			float4 _MainTex_ST;


			//パララックスマップ
			#ifdef _IS_PARALLAXMAP
			sampler2D _ParallaxTex_Back;
			float4 _ParallaxTex_Back_ST;
			sampler2D _ParallaxHeightMap_Back;
			float4 _ParallaxHeightMap_Back_ST;
			sampler2D _ParallaxNormalMap_Back;
			float4 _ParallaxNormalMap_Back_ST;
			sampler2D _ParallaxMaskTex_Back;
			float4 _ParallaxMaskTex_Back_ST;

			sampler2D _ParallaxTex_Front;
			float4 _ParallaxTex_Front_ST;
			sampler2D _ParallaxMaskTex_Front;
			float4 _ParallaxMaskTex_Front_ST;
			sampler2D _ParallaxNormalMap_Front;
			float4 _ParallaxNormalMap_Front_ST;
			sampler2D _ParallaxHeightMap_Front;
			float4 _ParallaxHeightMap_Front_ST;

			uniform fixed _HeightAdjustment_Back;
			uniform fixed4 _PseudoLight_Back;
			uniform fixed _HeightAdjustment_Front;
			uniform fixed4 _PseudoLight_Front;
			#endif

			#ifdef _IS_GRADATION
			//グラデーション
			sampler2D _GradationTex;
			float4 _GradationTex_ST;
			uniform fixed _GradationPower;
			uniform half4 _SubtractionColor;
			#endif

			#ifdef _IS_RIMLIGHT
			//リムライト
			sampler2D _RimMaskTex;
			float4 _RimMaskTex_ST;
			uniform half4 _RimColor;
			uniform fixed _RimWidth;
			uniform fixed _RimPower;
			uniform fixed _RimRearAdjustment;
			uniform fixed _RimBias;
			#endif

			#ifdef _IS_ANGELRING
			//天使の輪
			sampler2D _AngelRing_Sampler;
			float4 _AngelRing_Sampler_ST;
			uniform half4 _AngelRingColor;
			uniform fixed _AngelRingPower;
			uniform fixed _AngelRingAdjustment;
			uniform fixed _AngelRingOffsetX;
			uniform fixed _AngelRingOffsetY;
			uniform fixed _AngelRingUpperLimit;
			uniform fixed _AngelRingLowerLimit;
			uniform fixed _AngelRingMoveSpeed;
			#endif

			#ifdef _IS_SPECULARMASK
			sampler2D _SpecularMaskTex;
			float4 _SpecularMaskTex_ST;
			uniform half4 _SpecularColor;
			uniform fixed _SpecularRoughness;
			uniform fixed _FresnelReflectance;
			#endif

			//ライトカラー
			uniform half4 _LightColor0;

			#ifdef _USECUTOUT_ON
                fixed _Cutoff;
            #endif

            v2f vert (appdata v)
            {
                v2f o = (v2f)0;
                o.pos = UnityObjectToClipPos(v.vertex);
                o.normal = UnityObjectToWorldNormal(v.normal);
				o.eyeDir = normalize(WorldSpaceViewDir(v.vertex));

				#ifdef _IS_SPECULARMASK
				o.lightDir = WorldSpaceLightDir(v.vertex);
				#endif

				#ifdef _IS_PARALLAXMAP
				TANGENT_SPACE_ROTATION;
				o.objeyeDir = normalize(mul(rotation, ObjSpaceViewDir(v.vertex)));
				#endif

				#ifdef _IS_ANGELRING
				o.viewforward = mul((float3x3)unity_CameraToWorld, float3(0,0,1));
				#endif

				#ifdef _IS_SPECULARMASK
                // ハーフベクトルを求める
                o.halfDir = normalize(_WorldSpaceLightPos0.xyz + o.eyeDir);
				#endif

				o.uv = v.uv; // そのままのTexcoord

                return o;
            }

            fixed4 frag (v2f i) : SV_Target
            {

				half4 col = tex2D( _MainTex, TRANSFORM_TEX( i.uv, _MainTex ) );
				
				//パララックスマップ(視差マッピング)
				#ifdef _IS_PARALLAXMAP

					#ifdef _IS_PARALLAXMAP_BACK
						// ハイトマップをサンプリングしてUVをずらす
						fixed4 height_back = tex2D(_ParallaxHeightMap_Back, i.uv);
						float2 parallaxuv_back = -i.objeyeDir.xy * height_back.r * _HeightAdjustment_Back;
						//パララックステクスチャ
						half4 parallaxtex_back = tex2D( _ParallaxTex_Back, i.uv + parallaxuv_back);

						fixed3 normal = UnpackNormal(tex2D(_ParallaxNormalMap_Back, i.uv + parallaxuv_back));
						fixed4 diff = saturate(dot(normal, _PseudoLight_Back.xyz)) * _LightColor0;
				
						//マスクパララックス
						parallaxtex_back *= tex2D( _ParallaxMaskTex_Back, i.uv + parallaxuv_back).r;
						fixed MaskJudg_back = tex2D( _ParallaxMaskTex_Back, i.uv + parallaxuv_back).r;
						
						if(MaskJudg_back != 0) col.rgb *= parallaxtex_back.rgb * diff;
					#endif

					#ifdef _IS_PARALLAXMAP_FRONT
						// ハイトマップをサンプリングしてUVをずらす
						fixed4 height_front = tex2D(_ParallaxHeightMap_Front, i.uv);
						float2 parallaxuv_front = i.objeyeDir.xy * height_front.r * _HeightAdjustment_Front;
						//パララックステクスチャ
						half4 parallaxtex_front = tex2D( _ParallaxTex_Front, i.uv + parallaxuv_front);

						fixed3 normal_front = UnpackNormal(tex2D(_ParallaxNormalMap_Front, i.uv + parallaxuv_front));
						fixed4 diff_front = saturate(dot(normal_front, _PseudoLight_Front.xyz)) * _LightColor0;

						//マスクパララックス
						parallaxtex_front *= tex2D( _ParallaxMaskTex_Front, i.uv + parallaxuv_front).r;
						fixed MaskJudg_front = tex2D( _ParallaxMaskTex_Front, i.uv + parallaxuv_front).r;
				
						if(MaskJudg_front != 0) col.rgb += parallaxtex_front.rgb * diff;
					#endif
				#endif

				//グラデーションテクスチャーを減算
				#ifdef _IS_GRADATION
					half4 AddGradation = tex2D(_GradationTex, i.uv) * float4(_SubtractionColor.rgb, 0) * _GradationPower;
					
					col -= AddGradation;
				#endif

				//天使の輪
				#ifdef _IS_ANGELRING
					//ライトベクトルとカメラベクトルの内積
					half lightcameradot = clamp(-i.viewforward.y * _AngelRingMoveSpeed, _AngelRingLowerLimit, _AngelRingUpperLimit);
					//調整
					half2 f2_uvangel = i.uv + float2( _AngelRingOffsetX, _AngelRingOffsetY) - float2( 0, lightcameradot * _AngelRingAdjustment);

					half4 AddAngelRing = tex2D(_AngelRing_Sampler, f2_uvangel) * float4(_AngelRingColor.rgb, 0) *_AngelRingPower;

					//天使の輪を加算
					col += AddAngelRing;
				#endif

				//ライト色を反映
				col.rgb *= _LightColor0.rgb;

				//リムライト
				#ifdef _IS_RIMLIGHT
					
					half3 ViewNormalDir = normalize(mul((float3x3)UNITY_MATRIX_V, i.normal));
					half4 rimcol = tex2D( _RimMaskTex, i.uv);
					fixed rim = 0.0;
					fixed bias = 0.0; 

					#ifdef _IS_RIMBIAS
						bias = _RimBias;
					#endif

					//リムライトを裏側に描画しない
					if(dot(i.eyeDir, i.normal) >= -_RimRearAdjustment)
					{
						rim = 1.0 - saturate(dot (float3(i.eyeDir.x + bias, i.eyeDir.y, i.eyeDir.z), i.normal));
					}

					half4 color = 0;
					#ifdef _IS_LIGHTCOLOR
						//リムライトの色をディクショナルライトの色にする。						
						color.rgb = smoothstep(_RimWidth, _RimPower, rim) * _LightColor0.rgb * rimcol.r;
					#else
						//リムライトの色を指定色にする。
						color.rgb = smoothstep(_RimWidth, _RimPower, rim) * _LightColor0.rgb * _RimColor * rimcol.r;
					#endif

					col.rgb += color.rgb;
				#endif

				//スペキュラーマスク
				#ifdef _IS_SPECULARMASK
					half4 Spec = tex2D( _SpecularMaskTex, i.uv);

					float NdotH = dot(i.normal, i.halfDir);	//NormalVector dot HalfVector
					float NdotV = dot(i.normal, i.eyeDir);	//NormalVector dot ViewVector
					float NdotL = dot(i.normal, i.lightDir);//NormalVector dot LightVector
					float VdotH = dot(i.eyeDir, i.halfDir);	//ViewVector   dot HalfVector

					float alpha = _SpecularRoughness * _SpecularRoughness;
					float alpha2 = alpha * alpha;
					float t = ((NdotH * NdotH) * (alpha2 - 1.0) + 1.0);
					float PI = 3.1415926535897;
					float D = alpha2 / (PI * t * t);

					float F = pow(1.0 - VdotH, 5.0);
					F *= (1.0 - _FresnelReflectance);
					F += _FresnelReflectance;

					float NH2 = 2.0 * NdotH;
					float g1 = (NH2 * NdotV) / VdotH;
					float g2 = (NH2 * NdotL) / VdotH;
					float G = min(1.0, min(g1, g2));

					float specularReflection = (D * F * G) / (4.0 * NdotV * NdotL + 0.000001);

					col.rgb += (_SpecularColor.rgb * specularReflection) * Spec.r;
				#endif

				#ifdef _USECUTOUT_ON
                    clip( col.a - _Cutoff);
                #endif

				/// END ///			
                return col;
            }
            ENDCG
        }
    }
