Shader "Hidden/CharaShaderOutLine"
{
	Properties {
		[HideInInspector] _Mode ("__mode", Float) = 2.0
		
		[Enum(UnityEngine.Rendering.CullMode)]_CullMode("Cull Mode", Float) = 2
		[Enum(UnityEngine.Rendering.CompareFunction)]_ZTestMode("ZTest Mode", Float) = 4
		[Toggle]_ZWriteParam("ZWrite", Float) = 1
		[HideInInspector] _BlendMode ("__blendmode", Float) = 0.0
		[Enum(UnityEngine.Rendering.BlendMode)]_SrcBlend("Blend Src", Float) = 5
		[Enum(UnityEngine.Rendering.BlendMode)]_DstBlend("Blend Dst", Float) = 10

		_StencilRef("StencilRef", Int) = 0

		[Toggle] _Is_Shadow("Is_Shadow", Float) = 0

		_Cutoff("Alpha Cutoff", Range(0.0, 1.0)) = 0.5
		[HideInInspector][Toggle] _UseCutout("__useCutout", Float) = 0

		_MainTex ("Main", 2D) = "black" {}

		//パララックスマップ
		[Toggle(_IS_PARALLAXMAP)]
		_Is_ParallaxMap ("_Is_ParallaxMap", Float ) = 0
		[Toggle(_IS_PARALLAXMAP_BACK)]
		_Is_ParallaxMap_Back ("_Is_ParallaxMap_Back", Float ) = 0
		_ParallaxTex_Back ("ParallaxTex_Back", 2D) = "white" {}

		_ParallaxNormalMap_Back ("ParallaxNormalMap_Back", 2D) = "white" {}
		_ParallaxHeightMap_Back ("ParallaxHeightMap_Back", 2D) = "white" {}
		_HeightAdjustment_Back ("HeightAdjustment_Back", Range(0.0, 0.5)) = 0.02
		_PseudoLight_Back ("PseudoLight_Back", Vector) = (0, 0, 0)
		[Toggle(_IS_PARALLAXMAP_FRONT)]
		_Is_ParallaxMap_Front ("_Is_ParallaxMap_Front", Float ) = 0
		_ParallaxTex_Front ("ParallaxTex_Front", 2D) = "white" {}
		_ParallaxMaskTex_Front ("ParallaxMaskTex_Front", 2D) = "white" {}
		_ParallaxNormalMap_Front ("ParallaxNormalMap_Front", 2D) = "white" {}
		_ParallaxHeightMap_Front ("ParallaxHeightMap_Front", 2D) = "white" {}
		_HeightAdjustment_Front ("HeightAdjustment_Front", Range(0.0, 0.5)) = 0.02
		_PseudoLight_Front ("PseudoLight_Front", Vector) = (0, 0, 0)

		//グラデーション
		[Toggle(_IS_GRADATION)]
		_Is_Gradation ("Is_Gradation", Float ) = 0
		_GradationTex("Gradation", 2D) = "black" {}
		_GradationPower ("GradationPower", Range(0, 10)) = 1
		_SubtractionColor("SubtractionColor", Color) = (0, 0, 0, 0)

		//リムライト
		[Toggle(_IS_RIMLIGHT)]
		_Is_RimLight ("Is_RimLight", Float ) = 0
		_RimMaskTex ("RimMaskTex", 2D) = "white" {}
		[Toggle(_IS_USEVERTEXCOLOR_RED)]
		_Is_UseVertexColor_Red ("_Is_UseVertexColor_Red", Float ) = 1
		[Toggle(_IS_RIMLIGHT_USEVERTEXCOLOR_GREEN)]
		_Is_RimLight_UseVertexColor_Green ("_Is_RimLight_UseVertexColor_Green", Float ) = 1
		_Rim_Red_Threshold ("_Rim_Red_Threshold", Range(0, 1)) = 0.5
		_Rim_Green_Threshold ("_Rim_Green_Threshold", Range(0, 1)) = 0.5
		_Rim_Blue_Threshold ("_Rim_Blue_Threshold", Range(0, 1)) = 0.5
		_Rim_Alpha_Threshold ("_Rim_Alpha_Threshold", Range(0, 1)) = 0.5
		[Toggle(_IS_LIGHTCOLOR)]
		_Is_LightColor ("Is_LightColor", Float ) = 1
		_RimColor ("RimColor", Color) = (1, 1, 1, 1)
		_RimWidth("RimWidth", Range(0, 1)) = 0
		_RimPower("RimPower", Range(0, 1)) = 1
		_RimRearAdjustment("RimRearAdjustment", Range(0, 1)) = 0.1

		_AngelRingLowerLimit("AngelRingLowerLimit", Range(-1, 1)) = -1

		_Is_SpecularMask ("_Is_SpecularMask", Float ) = 0
		_SpecularMaskTex ("SpecularMaskTex", 2D) = "white" {}

		_SpecularRoughness("_SpecularRoughness", Range(0, 1)) = 0.5
		_FresnelReflectance("FresnelReflectance", Range(0, 1)) = 0.5

		//アニメ線
		[Toggle(_IS_ANIMELINE)]
		_Is_AnimeLine ("_Is_AnimeLine", Float ) = 0
		[Toggle(_IS_ANIMELINE_USEVERTEXCOLOR_GREEN)]
		_Is_AnimeLine_UseVertexColor_Green ("_Is_AnimeLine_UseVertexColor_Green", Float ) = 1
		_AnimeLine_Red_Threshold ("_AnimeLine_Red_Threshold", Range(0, 1)) = 0.5
		_AnimeLine_Green_Threshold ("_AnimeLine_Green_Threshold", Range(0, 1)) = 0.5
		_AnimeLine_Blue_Threshold ("_AnimeLine_Blue_Threshold", Range(0, 1)) = 0.5
		_AnimeLine_Alpha_Threshold ("_AnimeLine_Alpha_Threshold", Range(0, 1)) = 0.5
		_AnimeLineColor ("_AnimeLineColor", Color) = (1, 1, 1, 1)
		_AnimeLine_Adjustment ("_AnimeLine_Adjustment", Range(0, 1)) = 1
		[Enum(Addition,0,Subtraction,1)] _AnimeLineArithmetic ("_AnimeLineArithmetic", Float) = 0

		_OutlineMaskTex ("OutlineMaskTex", 2D) = "white" {}
		_OutlineMainColor ("OutlineMainColor", Color) = (.5,.5,.5,1)
		_OutlineColor ("OutlineColor", Color) = (0, 0, 0, 1)
		_Outline  ("Outline", float) = .005
		_OutlineMax("OutlineMax", Range(0.0, 10.0)) = .1
		[Toggle(OUTLINEZWRITE)]_OutlineZWrite("OutlineZWrite", Float) = 1

		//頂点カラー頬染め
		//一時オンオフ切り替え無し、オフにすると描画優先がかわってしまう？
		[Toggle(_IS_VERTEXCOLORMULTI)]
		_Is_VertexColorMulti ("_Is_VertexColorMulti", Float ) = 0
		_VertexColor ("VcolColor", Color ) =  (1, 1, 1, 1)
		_VcolPower("VcolPower", Range(0, 1)) = 1
		
		//瞳ウルウル
		[HideInInspector]_Flat ("_Flat", Range(0, 1)) = 1
		[HideInInspector]_Scale ("_Scale", Range(0, 0001)) = 0		
	}
	SubShader
	{	
		Tags { "Queue"="Transparent" "RenderType"="Transparent" "IgnoreProjector"="True" "LightMode"="ForwardBase" }
		Cull [_CullMode]
		ZTest [_ZTestMode]
		ZWrite [_ZWriteParam]
		Blend [_SrcBlend] [_DstBlend]

		//パス1：アウトライン
		UsePass "Hidden/OutLine/OUTLINE"
		
		//パス2：天使の輪、リムライト、グラデーション
		UsePass "Hidden/CharaShader/BASE"
	}
	CustomEditor "CharaShaderGUI"
}
