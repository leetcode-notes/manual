import os
import sys
sys.path.append(os.pardir)

import shutil
from datetime import datetime
import tkinter as tk
import tkinter.messagebox
from enum import Enum, auto

import tkinter, tkinter.filedialog
from PIL import Image

import numpy as np
import cv2

from skimage import data, io, filters

""" グローバル変数の定義
"""
entry = None
response_area = None
lb = None
action = None
on_canvas = None
log = []
question = 0

FACE_ICON_FILE = 'smile.png'
WM_FILE = 'watermark.png'
CASCADE_FILE = 'haarcascade_frontalface_default.xml'

def face_to_smile():
    import fn_smile_watermark
    fn_smile_watermark.smile_watermark(FACE_ICON_FILE, WM_FILE, CASCADE_FILE)

def watermark():
    import fn_watermark
    fn_watermark.watermark(WM_FILE)

def sobel():
    import fn_sobel
    fn_sobel.sobel(WM_FILE)

def facerecognition():
    import fn_facerecognition
    fn_facerecognition.facerecognition(WM_FILE)

def color_binarization():
    import fn_color_binarization
    fn_color_binarization.color_binarization(WM_FILE)

def run():
    # グローバル変数を使用するための記述
    global entry, response_area, lb, action, canvas, on_canvas, ptyna_images

    # メインウィンドウを作成
    root = tk.Tk()
    # ウィンドウのサイズを設定
    root.geometry('400x140')
    # ウィンドウのタイトルを設定
    root.title('Intelligent Agent : ')
    # フォントの用意
    font = ('Helevetica', 14)
    font_log = ('Helevetica', 11)

    def callback():
        """ 終了時の処理
        """
        # メッセージボックスの[OK]ボタンクリック時の処理
        if tkinter.messagebox.askyesno(
                'Quit?', '終了しますか?'):
            root.destroy()
        # [キャンセル]ボタンクリック
        else:
            root.destroy()

    root.protocol('WM_DELETE_WINDOW', callback)

    # メニューバーの作成
    menubar = tk.Menu(root)
    root.config(menu=menubar)
    # 「ファイル」メニュー
    filemenu = tk.Menu(menubar)
    menubar.add_cascade(label='ファイル', menu=filemenu)
    filemenu.add_command(label='閉じる', command=callback)
    # 「オプション」メニュー
    action = tk.IntVar()
    optionmenu = tk.Menu(menubar)
    menubar.add_cascade(label='オプション', menu=optionmenu)
    optionmenu.add_radiobutton(
        label='Responderを表示',  # アイテム名
        variable=action,  # 選択時の値を格納するオブジェクト
        value=0  # actionの値を0にする
    )
    optionmenu.add_radiobutton(
        label='Responderを表示しない',  # アイテム名
        variable=action,  # 選択時の値を格納するオブジェクト
        value=1  # actionの値を0にする
    )

    frame = tk.Frame(
        root,
        relief=tk.RIDGE,
        borderwidth=4
    )
    button = tk.Button(
        root,
        width=50,
        text='watermark+face to smile',
        command=face_to_smile
    )
    button.pack()
    frame.place(x=20, y=20)

    button2 = tk.Button(
        root,
        width=50,
        text='watermark',
        command=watermark
    )
    button2.pack()
    frame.place(x=20, y=80)

    button2 = tk.Button(
        root,
        width=50,
        text='輪郭検出',
        command=sobel
    )
    button2.pack()
    frame.place(x=20, y=80)

    button2 = tk.Button(
        root,
        width=50,
        text='カラー2値化',
        command=color_binarization
    )
    button2.pack()
    frame.place(x=20, y=80)

    button2 = tk.Button(
        root,
        width=50,
        text='顔検出機能',
        command=facerecognition
    )
    button2.pack()  # フレームに左詰めで配置する
    frame.place(x=20, y=80)  # フレームを画面上に配置

    # メインループ
    root.mainloop()


if __name__ == '__main__':
    run()
