gray = 0.2126 * r + 0.7152 * g + 0.0722 * b

In [5]:
# sobel Filter
K_size = 3
In [6]:
## Zero padding
pad = K_size // 2
out = np.zeros((H + pad*2, W + pad*2), dtype=np.float)
out[pad:pad+H, pad:pad+W] = gray.copy().astype(np.float)
tmp = out.copy()
In [7]:
## Laplacian vertical
K = [[0., 1., 0.],[1., -4., 1.], [0., 1., 0.]]
In [8]:

    for x in range(W):
        out[pad+y, pad+x] = np.mean(K * (tmp[y:y+K_size, x:x+K_size]))
