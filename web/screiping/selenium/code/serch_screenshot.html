selenium serch_screenshot¶

# -*- encoding: utf-8 -*-
# pip install cssselect
from selenium import webdriver
import lxml.html

chromedriver = r"chromedriver.exe"
browser = webdriver.Chrome(executable_path=chromedriver)
# Windowサイズを最大に設定
browser.maximize_window()
browser.get('http://www.yahoo.co.jp/')

browser.save_screenshot('01_yahooTop.png')
# タイトル表示
print('------------------------------')
print(browser.title)
# HTMLソース取得
root = lxml.html.fromstring(browser.page_source)
# リンク文字列を取得(CSSPATH指定時)
print('------------------------------')
cp_text = root.cssselect('#topicsfb > div.topicsindex > ul.emphasis > li:nth-child(1) > a')[0].text_content()
print(cp_text)

print('------------------------------')
# リンク文字列を取得(XPATH指定時)
xp_text = root.xpath('//*[@id="topicsfb"]/div[1]/ul[1]/li[2]/a')[0].text
print(xp_text)

# テキストボックスに文字列を入力して検索ボタンを押下
input_box = browser.find_element_by_id('srchtxt')
input_box.send_keys('検索テスト')
browser.find_element_by_id('srchbtn').click()

browser.save_screenshot('02_searchResult.png')
