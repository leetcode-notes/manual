using UnityEngine;
using UnityEngine.SceneManagement;
#if UNITY_EDITOR
using UnityEditor;
#endif
using System.Collections;
using System.Collections.Generic;


namespace Gluon
{
    public class EffectChecker : MonoBehaviour
    {
        enum Mode
        {
            DEFAULT,
            SIMULATE,
            FBF
        }

        Mode mode = Mode.DEFAULT;

        [SerializeField]
        GameObject _target = null;

        GameObject cloneObject = null;
        bool isPause = false;

        bool isAutoRemove = false;

        float defaultScale = 1f;

        float updateSpeed = 0.0f;
        float simulateSpeed = 1.0f;

        void Start()
        {
            isPause = false;
            updateSpeed = 0;
        }

        void Update()
        {
            if (cloneObject == null)
            {
                return;
            }

            if (isAutoRemove)
            {
                if (ConfirmStopedParticles(cloneObject))
                {
                    Destroy(cloneObject);
                    isPause = false;
                }
            }

            if (mode != Mode.DEFAULT)
            {
                UpdateParticles(cloneObject);
            }
        }
        void LateUpdate()
        {
            if (mode == Mode.FBF && cloneObject != null)
            {
                PauseParticles(cloneObject);
                isPause = true;
                mode = Mode.DEFAULT;
            }
        }

        void PlayEffect()
        {
            if (_target != null)
            {
                if (cloneObject != null)
                {
                    StopParticles(cloneObject);
                    PlayParticles(cloneObject);
                    return;
                }

                GameObject clone = Instantiate<GameObject>(_target);
                clone.name = clone.name.Remove(clone.name.IndexOf("(Clone)"));
                clone.transform.parent = transform;
                clone.transform.localPosition = Vector3.zero;
                clone.transform.localScale = Vector3.one;
                cloneObject = clone;

                isPause = false;
            }
        }
        void PlayEffect(GameObject target)
        {
            if (target != null)
            {
                if (cloneObject != null)
                {
                    StopParticles(cloneObject);
                    PlayParticles(cloneObject);
                    return;
                }

                GameObject clone = Instantiate<GameObject>(target);
                clone.name = clone.name.Remove(clone.name.IndexOf("(Clone)"));
                clone.transform.parent = transform;
                clone.transform.localPosition = Vector3.zero;
                clone.transform.localScale = Vector3.one;
                cloneObject = clone;

                if(defaultScale != 1f)
                {
                    SetupParticleScale(cloneObject, defaultScale);
                }

                isPause = false;
            }
        }
        void PauseEffect()
        {
            if (cloneObject == null)
            {
                return;
            }
            if (!isPause)
            {
                PauseParticles(cloneObject);
                isPause = true;
            }
            else
            {
                PlayParticles(cloneObject);
                isPause = false;
            }
        }
        void AddFrameEffect()
        {
            if (cloneObject == null)
            {
                return;
            }

            PauseParticles(cloneObject);
            isPause = true;

            mode = Mode.FBF;
            updateSpeed = 1f / ClaculateFrameRate();
        }
        void StopEffect()
        {
            if (cloneObject == null)
            {
                return;
            }

            StopParticles(cloneObject);
            isPause = false;
        }
        void RemoveEffect()
        {
            Destroy(cloneObject);
            isPause = false;
        }

        void UpdateDefaultScale()
        {
            if(cloneObject == null)
            {
                return;
            }

            SetupParticleScale(cloneObject, defaultScale);
        }

        void SetupParticleScale(GameObject target, float scale)
        {
            ParticleSystem system = target.GetComponent<ParticleSystem>();
            if (system != null)
            {
                target.transform.localScale = Vector3.one * scale;
            }

            for (int index = 0; index < target.transform.childCount; index++)
            {
                SetupParticleScale(target.transform.GetChild(index).gameObject, scale);
            }
        }

        bool ConfirmStopedParticles(GameObject target)
        {
            ParticleSystem system = target.GetComponent<ParticleSystem>();
            if (system != null)
            {
                return system.isStopped;
            }

            for (int index = 0; index < target.transform.childCount; index++)
            {
                if (ConfirmStopedParticles(target.transform.GetChild(index).gameObject))
                {
                    return true;
                }
            }
            return false;
        }
        void PlayParticles(GameObject target)
        {
            ParticleSystem system = target.GetComponent<ParticleSystem>();
            if (system != null)
            {
                system.Play(true);
            }

            for (int index = 0; index < target.transform.childCount; index++)
            {
                PlayParticles(target.transform.GetChild(index).gameObject);
            }
        }
        void PauseParticles(GameObject target)
        {
            ParticleSystem system = target.GetComponent<ParticleSystem>();
            if (system != null)
            {
                system.Pause(true);
            }

            for (int index = 0; index < target.transform.childCount; index++)
            {
                PauseParticles(target.transform.GetChild(index).gameObject);
            }
        }
        void StopParticles(GameObject target)
        {
            ParticleSystem system = target.GetComponent<ParticleSystem>();
            if (system != null)
            {
                system.Stop(true);
            }

            for (int index = 0; index < target.transform.childCount; index++)
            {
                StopParticles(target.transform.GetChild(index).gameObject);
            }
        }
        void UpdateParticles(GameObject target)
        {
            ParticleSystem system = target.GetComponent<ParticleSystem>();
            if (system != null)
            {
                system.Simulate(updateSpeed, false, false);
            }

            for (int index = 0; index < target.transform.childCount; index++)
            {
                UpdateParticles(target.transform.GetChild(index).gameObject);
            }
        }
        float ClaculateFrameRate()
        {
            float frameRate = (float)Application.targetFrameRate;
            if (frameRate == -1f)
            {
                frameRate = 60f;
            }
            return frameRate;
        }

#if UNITY_EDITOR
        [CustomEditor(typeof(EffectChecker))]
        class EffectCheckerInspector : Editor
        {
            GUIStyle largeCategoryStyle = null;

            public override void OnInspectorGUI()
            {
                base.OnInspectorGUI();

                GUILayout.Space(10f);

                if(largeCategoryStyle == null)
                {
                    largeCategoryStyle = new GUIStyle(EditorStyles.boldLabel);
                    largeCategoryStyle.fontSize = 15;
                }

                EffectChecker checker = target as EffectChecker;

                GUILayout.Label("コントローラ", largeCategoryStyle);
                GUILayout.BeginHorizontal();
                GUILayout.Space(15f);
                GUILayout.Label("再生スピード", EditorStyles.label);
                float simulateSpeed = EditorGUILayout.FloatField(checker.simulateSpeed);
                if (checker.simulateSpeed != simulateSpeed)
                {
                    checker.simulateSpeed = simulateSpeed;
                    if (checker.simulateSpeed != 1f)
                    {
                        checker.updateSpeed = checker.simulateSpeed / checker.ClaculateFrameRate();
                        checker.mode = Mode.SIMULATE;
                    }
                    else
                    {
                        checker.mode = Mode.DEFAULT;
                        checker.PlayEffect();
                    }
                }
                GUILayout.EndHorizontal();

                GUILayout.BeginHorizontal();
                GUILayout.Space(15f);
                if (GUILayout.Button("再生"))
                {
                    checker.PlayEffect();
                }
                if (GUILayout.Button("一時停止"))
                {
                    checker.PauseEffect();
                }
                if (GUILayout.Button("停止"))
                {
                    checker.StopEffect();
                }
                GUILayout.EndHorizontal();
                GUILayout.BeginHorizontal();
                GUILayout.Space(15f);
                if (GUILayout.Button("コマ送り再生"))
                {
                    checker.PlayEffect();
                    checker.AddFrameEffect();
                }
                if (GUILayout.Button("コマ送り"))
                {
                    checker.AddFrameEffect();
                }
                GUILayout.EndHorizontal();

                GUILayout.Space(10f);

                GUILayout.Label("削除機能", largeCategoryStyle);
                GUILayout.BeginHorizontal();
                GUILayout.Space(15f);
                GUILayout.Label("自動削除機能", EditorStyles.label);
                checker.isAutoRemove = EditorGUILayout.Toggle(checker.isAutoRemove);
                GUILayout.EndHorizontal();
                GUILayout.BeginHorizontal();
                GUILayout.Space(15f);
                if (!checker.isAutoRemove)
                {
                    if (GUILayout.Button("削除"))
                    {
                        checker.RemoveEffect();
                    }
                }
                GUILayout.EndHorizontal();

                GUILayout.Space(10f);

                GUILayout.Label("拡張機能", largeCategoryStyle);
                GUILayout.BeginHorizontal();
                GUILayout.Space(15f);
                GUILayout.Label("倍率", EditorStyles.label);
                float defaultScale = EditorGUILayout.FloatField(checker.defaultScale);
                if (defaultScale != checker.defaultScale)
                {
                    checker.defaultScale = defaultScale;
                    if (checker.cloneObject != null)
                    {
                        checker.UpdateDefaultScale();
                    }
                }
                GUILayout.EndHorizontal();

                GUILayout.Space(10f);

                GUILayout.BeginHorizontal();
                GUILayout.Space(15f);
                if (GUILayout.Button("リソース反映"))
                {
                    ApplayAssets(checker);
                }
                GUILayout.EndHorizontal();
            }

            void ApplayAssets(EffectChecker checker)
            {
                if (checker._target == null || checker.cloneObject == null)
                {
                    return;
                }

                string assetPath = AssetDatabase.GetAssetPath(checker._target);
                checker._target = PrefabUtility.ReplacePrefab(checker.cloneObject, checker._target, ReplacePrefabOptions.ConnectToPrefab);
            }
        }
#endif
    }

#if UNITY_EDITOR
    class ParticleTextureView : EditorWindow
    {
        [MenuItem("Gluon/Effect/Open Texture Viewer")]
        static void ShowWindow()
        {
            ParticleTextureView window = EditorWindow.GetWindow<ParticleTextureView>("Tiles Texture View");
            window.Show();
        }

        GameObject target = null;

        ParticleSystemRenderer renderer = null;
        Texture2D texture = null;
        Texture2D gridTexture = null;

        struct SHEET_DATA
        {
            public int x, y;
            public int width, height;

            public SHEET_DATA(int x, int y, int width, int height)
            {
                this.x = x;
                this.y = y;

                this.width = width;
                this.height = height;
            }
        }
        List<SHEET_DATA> dataList = new List<SHEET_DATA>();

        int _tilesX = 1, _tilesY = 1;

        bool showNumbers = true;

        GUIStyle fontStyle = null;

        void OnGUI()
        {
            float offsetX = 0f, offsetY = 0f;

            const float COMMON_HEIGHT = 25f;

            GUILayout.BeginArea(new Rect(offsetX, offsetY, 500f, COMMON_HEIGHT));
            GUILayout.BeginHorizontal();
            GameObject obj = EditorGUILayout.ObjectField(target, typeof(GameObject), true) as GameObject;
            Texture2D tmpTexture = EditorGUILayout.ObjectField(texture, typeof(Texture2D), true) as Texture2D;
            GUILayout.EndHorizontal();
            GUILayout.EndArea();

            offsetX = 0f;
            offsetY += COMMON_HEIGHT;

            GUILayout.BeginArea(new Rect(offsetX, offsetY, 250f, COMMON_HEIGHT));
            string loadButtonText = target == null || renderer == null || target != renderer.gameObject ? "読み込み" : "再読み込み";
            GUILayout.BeginHorizontal();
            bool load = GUILayout.Button(loadButtonText);
            bool remove = GUILayout.Button("解放");
            GUILayout.EndHorizontal();
            GUILayout.EndArea();

            offsetX = 0f;
            offsetY += COMMON_HEIGHT;

            bool isChangedTilesNum = false;

            GUILayout.BeginArea(new Rect(offsetX, offsetY, 500f, COMMON_HEIGHT));
            GUILayout.BeginHorizontal();
            int tilesX = EditorGUILayout.IntField("Tiles X", _tilesX);
            tilesX = tilesX < 1 ? 1 : tilesX;
            isChangedTilesNum = tilesX != _tilesX;
            _tilesX = tilesX;

            int tilesY = EditorGUILayout.IntField("Tiles Y", _tilesY);
            tilesY = tilesY < 1 ? 1 : tilesY;
            isChangedTilesNum |= tilesY != _tilesY;
            _tilesY = tilesY;
            GUILayout.EndHorizontal();
            GUILayout.EndArea();

            offsetX = 0f;
            offsetY += COMMON_HEIGHT + 10f;

            if (remove)
            {
                target = null;
                renderer = null;

                tmpTexture = null;
                texture = null;
                DestroyImmediate(gridTexture);
            }

            if(tmpTexture != null &&
                tmpTexture != texture)
            {
                texture = tmpTexture as Texture2D;
                isChangedTilesNum = true;

                target = null;
                obj = null;
            }

            if (load && obj != null)
            {
                if (target == null ||
                    renderer == null ||
                    target != renderer.gameObject)
                {
                    texture = null;

                    ParticleSystemRenderer r = obj.GetComponent<ParticleSystemRenderer>();

                    if (r != null &&
                        r != renderer)
                    {
                        if (r.material != null)
                        {
                            texture = r.material.mainTexture as Texture2D;
                        }
                        else if (r.sharedMaterial != null)
                        {
                            texture = r.sharedMaterial.mainTexture as Texture2D;
                        }

                        isChangedTilesNum = true;
                    }

                    renderer = r;
                }
            }

            if (isChangedTilesNum &&
                texture != null)
            {
                dataList.Clear();

                if (gridTexture == null)
                {
                    gridTexture = new Texture2D(texture.width, texture.height, TextureFormat.RGBA32, false, true);
                }
                else if (gridTexture.width != texture.width || gridTexture.height != texture.height)
                {
                    gridTexture.Resize(texture.width, texture.height);
                }


                int dx = gridTexture.width / _tilesX;
                int dy = gridTexture.height / _tilesY;

                for (int y = 0; y < gridTexture.height; y++)
                {
                    for (int x = 0; x < gridTexture.width; x++)
                    {
                        Color color = new Color(0f, 0f, 0f, 0f);
                        if (x % dx == 0 || y % dy == 0)
                        {
                            color = Color.green;
                        }
                        gridTexture.SetPixel(x, y, color);

                        if (x == 0 && y % dy == 0)
                        {
                            dataList.Add(new SHEET_DATA(x, y, dx, dy));
                        }
                        else if (y == 0 && x % dx == 0)
                        {
                            dataList.Add(new SHEET_DATA(x, y, dx, dy));
                        }
                    }
                }
                gridTexture.Apply();
            }

            target = obj;

            if (texture != null)
            {
                GUIContent texContent = new GUIContent(texture);
                Rect rect = GUILayoutUtility.GetRect(texContent, GUI.skin.label, GUILayout.ExpandWidth(true), GUILayout.ExpandHeight(true));
                rect.y += offsetY;

                GUI.Label(rect, texContent);
                GUI.Label(rect, new GUIContent(gridTexture));

                if(fontStyle == null)
                {
                    fontStyle = new GUIStyle(GUI.skin.label);
                }
                Rect labelRect;
                string labelText = "";

                float rectWidth = rect.width < texture.width ? rect.width : texture.width;
                float rectHeight = rectWidth * texture.height / texture.width;
                float onceWidth = rectWidth / _tilesX, onceHeight = rectHeight / _tilesY;

                fontStyle.fontSize = (int)onceWidth / 3;

                for (int index = 0; index < dataList.Count; index++)
                {
                    int gridX = dataList[index].x / dataList[index].width;
                    int gridY = dataList[index].y / dataList[index].height;

                    float posX = rect.x + onceWidth * gridX;
                    float posY = rect.y + onceHeight * gridY;

                    labelRect = new Rect(posX, posY, onceWidth, onceHeight);

                    if (dataList[index].x == 0)
                    {
                        labelText = string.Format("{0}", gridY);
                    }
                    else if (dataList[index].y == 0)
                    {
                        labelText = string.Format("{0}", gridX);
                    }

                    GUI.Label(labelRect, labelText, fontStyle);
                }
            }
        }
    }
