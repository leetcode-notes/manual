EXAMPLE OF 3D BAR PLOT¶

  • The above graph used the bar method with 3 parameters; x, y, and z.
  • bar3d method takes 6 parameters; x, y, z, dx, dy, and dz.
In [1]:
import matplotlib.pyplot as plt, numpy as np
from mpl_toolkits.mplot3d import Axes3D
In [2]:
x = np.array(range(0, 3), float)
y = x.copy()
xpos, ypos = np.meshgrid(x, y)

z = np.random.rand(3,3)
z = np.array([[0.4, 0.5, 0.4],
             [1.5, 0.1, 0.8],
             [0.2, 0.2, 0.2]])

xpos = xpos.flatten()
ypos = ypos.flatten()
colors = ["b", "g", "crimson"]*3
zpos = np.zeros_like(xpos)
dx = 0.5* np.ones_like(zpos)
dy = dx.copy()
dz = z.flatten()
fig = plt.figure(figsize = (7, 5))
ax = fig.add_subplot(111, projection = "3d")
ax.bar3d(xpos, ypos, zpos, dx, dy, dz, color = colors)
plt.show()

%matplotlib nbagg
x = np.array(range(3), float)
y = x.copy()
xpos, ypos = np.meshgrid(x, y)
z = np.array([[0.23, 0.21, 0.3],
             [0.4, 0.5, 0.6],
             [0.7, 0.6, 0.9]])
colors = ["b", "g", "crimson"]*3
xpos = xpos.flatten()
ypos = ypos.flatten()
zpos = np.zeros_like(xpos)
dx = 0.5*np.ones_like(zpos)
dy  = dx.copy()
dz = z.flatten()

fig = plt.figure()
ax = fig.add_subplot(111, projection = "3d")
labels = ["Harvard", "Yale", "Berkeley"]
ax.set_title("University Salaries", fontsize = 20)
ax.set_xticklabels([3,2,1])
ax.set_xticks(range(3))
ax.set_xlabel("Ranking", labelpad = 7)

ax.set_yticks(range(3))
ax.set_yticklabels(["Admin", "Security", "Lecturer"])
ax.set_ylabel("Profession", labelpad= 10)

ax.set_zticklabels([0, 20000, 40000, 60000, 80000])
ax.set_zticks(np.arange(0.0, 1.0, 0.2))
ax.set_zlabel("Salary/$")

x_leg = plt.Rectangle((0,0), 1,1, fc = "b")
y_leg = plt.Rectangle((0, 0), 1,1, fc = "g")
z_leg = plt.Rectangle((0, 0), 1,1, fc = "crimson")
ax.legend([x_leg, y_leg, z_leg], labels, fontsize = 12, loc = "best")

ax.bar3d(xpos, ypos, zpos, dx, dy, dz, color = colors)
plt.show()
