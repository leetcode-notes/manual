from matplotlib.backends.backend_pdf import PdfPages as pdf

with pdf("C:/Users/Michael/Desktop/MPL_Graphs.pdf") as image:
    
    for i in range(3):
        
        fig = plt.figure()
        plt.plot(np.random.randint(1, 50, 12),
                color = ["b", "orange", "m"][i], lw = 7)
        plt.title("Graph_{}".format(i))
        
        image.savefig(fig, bbox_inches = "tight", dpi = 300)
        
        plt.show();
