x = np.random.randint(1, n, n)
y = np.random.randint(1, n, n)

col = ["r", "g", "lightblue", "pink"]

plt.figure(figsize=(7, 5))
for a, b in  zip(x, y):
    
    if a >=  50 and b >= 50:
        
        plt.scatter(a, b, s = 400, c = col[0], marker = "s")
        
    elif a < 50  and a > 10 and b < 50 and b > 10:
        
        plt.scatter(a, b, s = 200, c = col[1], marker = "^")
        
    else:
        plt.scatter(a, b, s= 100, c = col[2], marker = "o")
        
plt.grid(True)
plt.show();
