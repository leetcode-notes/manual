import matplotlib.pyplot as plt, numpy as np
from mpl_toolkits.mplot3d import Axes3D

fig = plt.figure(figsize = (7,5))
ax1 = fig.add_subplot(121, projection ="3d")

colors = ["r", "b", "g", "m", "c", "orange"]

for col, z in zip(colors[:3], [3,2,1]):
    
    xs = np.arange(20) ; ys = np.random.rand(20)
    
    cs = [col] * len(xs)
    
    ax1.bar(xs, ys, zs = z, zdir="y", color = cs, alpha = 0.8)
ax1.set_title("First 3D Graph", fontsize =14)
ax1.set_xlabel("X", fontsize = 15)
ax1.set_ylabel("Y", fontsize = 15)
ax1.set_zlabel("Z", fontsize = 15)
ax1.legend(["x", "y", "z"], fontsize = 15, title = "colors", loc = (1.1,0.5))
ax1.set_xticklabels(["a", "b", "c", "d", "e"])
plt.savefig("3d_bar1.png", bbox_inches = "tight", dpi = 100)

plt.tight_layout()
plt.show()
