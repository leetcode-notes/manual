dictionary¶

内包表記¶

a = ["a", "b", "c"]
b = ['d', 'e', 'f']
dic = {}
for m, n in zip(a, b):
    dic[m] = n
print(dic)

d = {m: n for m, n in zip(a, b)}
print(d)

順番変わらない¶

from collections import OrderedDict

input_lines = input().split(" ")
dic = OrderedDict()
for m in input_lines:
    if not m in dic:
        dic[m] = 1
    else:
        dic[m] += 1
for value,item in dic.items():
    print(value,item)

辞書の数を更新¶

# 文房具ごとの在庫数
items = {'ボールペン': 12,
         'ノート': 3,
         'のり': 5}
item = 'ボールペン'
items[item] -= 5

for item, num in items.items():
    print(item, num)

辞書をループで処理する¶

enemies = {"ザコ":"スライム", "中ボス":"ドラゴン", "ラスボス":"魔王"}
print(enemies)
print(enemies["中ボス"])

for rank in enemies:
    print(enemies[rank] + "が、あらわれた！")
for (rank, enemy) in enemies.items():
    print(rank + "の" + enemy + "が、あらわれた！")

辞書のキー（key）の最大値・最小値を取得¶

d = {'a': 100, 'b': 20, 'c': 50, 'd': 100, 'e': 80}

max_v = max(d.values())
print(max_v)
# 100

min_v = min(d.values())
print(min_v)
# 20

辞書の値（value）の最大値・最小値を取得¶

max_v = max(d.values())
print(max_v)
# 100
source: dict_value_max_min.py

min_v = min(d.values())
print(min_v)
# 20

辞書の値が最大・最小となるキーを取得¶

max_k = max(d, key=d.get)
print(max_k)
# a

min_k = min(d, key=d.get)
print(min_k)
# b

辞書の値が最大・最小となるキーと値を同時に取得¶

max_kv = max(d.items(), key=lambda x: x[1])
print(max_kv)
# ('a', 100)

print(type(max_kv))
# <class 'tuple'>

タプルの　unpack を利用して、それぞれ別々の変数に代入することもできる。¶

max_k, max_v = max(d.items(), key=lambda x: x[1])
print(max_k)
# a

print(max_v)
# 100
