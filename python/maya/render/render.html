Aspect Ratio¶

rewd = cmds.textFieldGrp(self.rewd, q=True, tx=True)
reht = cmds.textFieldGrp(self.reht, q=True, tx=True)
device_aspect = float(float(rewd) * 1.0) / float(reht)
###1.0 = pixelAspect####
cmds.setAttr("defaultResolution.lockDeviceAspectRatio", 1)
cmds.setAttr("defaultResolution.width", int(rewd))
cmds.setAttr("defaultResolution.height", int(reht))
cmds.setAttr("defaultResolution.deviceAspectRatio", device_aspect)
