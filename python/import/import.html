default dict if文を書かなくてもいい¶

s = "fjdsjlfjadjffffjljkjgeio"

d = {}

for c in s:
    if not c in d:
        d[c] = 0
    d[c] += 1
print(d)

d = {}
for c in s:
    d = setdefault(c,0)
    d[c] += 1

from collections import defaultdict

d = defaultdict(int)

for c in s:
    d[d] += 1

print(d["f"])
