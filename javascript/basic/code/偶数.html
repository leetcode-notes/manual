code 偶数¶

process.stdin.resume();
process.stdin.setEncoding('utf8');
//標準入力から数字が入力されるので1からその数字までの
//偶数を出力して下さい。

var input_string = '';
process.stdin.on('data', function(chunk) {
  input_string += chunk;
});

process.stdin.on('end', function() {
  var num = input_string.replace(/(^ +)|( +$)/g, "");
  var i = 0;
  while(i <= num){
    i++;
    if( i%2 == 0) {
      console.log(i);
    }
  }
});
